142class PagesEndpoint(Endpoint): 

143 def create(self, **kwargs: Any) -> SyncAsync[Any]: 

144 """Create a new page in the specified database or as a child of an existing page. 

145 

146 If the parent is a database, the `properties` parameter must conform to the 

147 parent database's property schema. 

148 

149 If the parent is a page, the only valid property is `title`. The new page may 

150 include page content, described as blocks in the `children` parameter. 

151 """ 

152 return self.parent.request( 

153 path="pages", 

154 method="POST", 

155 body=pick(kwargs, "parent", "properties", "children"),
