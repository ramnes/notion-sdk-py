28@dataclass 

29class ClientOptions: 

30 """Options to configure the client. 

31 

32 Attributes: 

33 auth: Bearer token for authentication. If left undefined, the `auth` parameter 

34 should be set on each request. 

35 timeout_ms: Number of milliseconds to wait before emitting a 

36 `RequestTimeoutError`. 

37 base_url: The root URL for sending API requests. This can be changed to test with 

38 a mock server. 

39 log_level: Verbosity of logs the instance will produce. By default, logs are 

40 written to `stdout`. 

41 logger: A custom logger. 

42 notion_version: Notion version to use. 

43 """

112 def _parse_response(self, response: Response) -> Any: 

113 try: 

114 response.raise_for_status() 

115 except httpx.TimeoutException: 

116 raise RequestTimeoutError() 

117 except httpx.HTTPStatusError as error: 

118 body = error.response.json() 

119 code = body.get("code") 

120 if code and is_api_error_code(code): 

121 raise APIResponseError(response, body["message"], code) 

122 raise HTTPResponseError(error.response) 

123 

124 return response.json()
