14class RequestTimeoutError(Exception): 

15 """Request timed out. 

16 

17 The request that we made waits for a specified period of time or maximum number 

18 of retries to get the response. 

19 But if no response comes within the limited time or retries, 

20 then this Exception is raised. 

21 """

23 code = "notionhq_client_request_timeout"
